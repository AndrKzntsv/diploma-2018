\tikzset{
  boat/.pic={
    code={
      \coordinate (bow) at (0,0);
      \coordinate (stern) at (4,-1.25);

      \coordinate (stern_left_top) at ($ (stern) + (-0.75,-0.25) $);
      \coordinate (stern_left_bottom) at ($ (stern_left_top) + (0,-0.5) $);
      \coordinate (stern_right_top) at ($ (stern) + (0.75,0.25) $);
      \coordinate (stern_right_bottom) at ($ (stern_right_top) + (0,-0.5) $);

      \draw (bow) -- (stern);
      \draw (stern_left_top) -- (stern_right_top);
      \draw (stern_left_top) -- (stern_left_bottom);
      \draw (stern_right_top) -- (stern_right_bottom);

      \draw (stern_right_bottom) to[out=-125,in=-15,looseness=0.8] (stern_left_bottom);

      \draw (bow) to[out=-45,in=-190,looseness=0.8] (stern_left_top);
      \draw (bow) to[out=-75,in=-190,looseness=0.8] (stern_left_bottom);
      \draw (bow) to[out=0,in=160,looseness=0.5] (stern_right_top);

      \coordinate (mast_bottom) at ($ (bow)!0.6!(stern) $);
      \coordinate (mast_bottom_left) at ($ (mast_bottom) + (-0.05,0.03) $);
      \coordinate (mast_bottom_right) at ($ (mast_bottom) + (0.05,-0.03) $);
      \coordinate (mast_top) at ($ (mast_bottom) + (0,3) $);
      \coordinate (mast_top_left) at ($ (mast_top) + (-0.05,0.03) $);
      \coordinate (mast_top_right) at ($ (mast_top) + (0.05,-0.03) $);

      % \draw (mast_bottom) -- (mast_top);
      \draw (mast_bottom_left) -- (mast_top_left);
      \draw (mast_bottom_right) -- (mast_top_right);
    }
  }
}
